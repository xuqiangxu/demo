    int cross(vector<int> & p, vector<int> & q, vector<int> & r) {
        return (q[0] - p[0]) * (r[1] - q[1]) - (q[1] - p[1]) * (r[0] - q[0]);    //叉乘实现
    }

    vector<vector<int>> outerTrees(vector<vector<int>>& trees) {
        int n = trees.size();
        if (n < 4) {
            return trees;		//三个点必构成凸包
        }

        int leftMost = 0;	//找到最左边的点
        for (int i = 0; i < n; i++) {
            if (trees[i][0] < trees[leftMost][0]) {     
                leftMost = i; 
            }
        }

        vector<vector<int>> res;   //结果储存
        vector<bool> visit(n, false);  //判断访问与否
        int p = leftMost;     //首位
        do {
            int q = (p + 1) % n;          //从leftmost开始遍历
            for (int r = 0; r < n; r++) {
                /* 如果 r 在 pq 的右侧，则 q = r */ 
                if (cross(trees[p], trees[q], trees[r]) < 0) {
                    q = r;
                }
            }
            /* 是否存在点 i, 使得 p 、q 、i 在同一条直线上 */
            for (int i = 0; i < n; i++) {
                if (visit[i] || i == p || i == q) {
                    continue;
                }
                if (cross(trees[p], trees[q], trees[i]) == 0) {
                    res.emplace_back(trees[i]);
                    visit[i] = true;
                }
            }
            if  (!visit[q]) {
                res.emplace_back(trees[q]);
                visit[q] = true;
            }
            p = q;
        } while (p != leftMost);
        return res;
    }
