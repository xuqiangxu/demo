string str1,str2;
int m = str1.size(), n = str2.size();
vector<vector<int>> f(m+1,vector<int>(n+1,0))
//最好是这样 每次添加过就不用继续添加了
for(int i=0; i<str1.size(); ++i) {

	for(int j=0; j<str2.size(); ++j) {
		//怎么纪录非公共子串呢
		if(str1[i] == str2[j]) {
			f[i+1][j+1]  = f[i][j] + 1；
		}
		else {
			f[i+1][j+1] = max(f[i][j+1],f[i+1][j]);
		}		
	}

}
return f[m][n];
